CREATE TABLE ACCOUNT (id BIGINT PRIMARY KEY AUTO_INCREMENT, saldo DECIMAL, number VARCHAR2(2000), idClient BIGINT)

CREATE TABLE DOCUMENT (id BIGINT PRIMARY KEY AUTO_INCREMENT, accDT BIGINT, accCT BIGINT, summa DECIMAL, purpose VARCHAR2(2000), date DATE)

CREATE TABLE CLIENT (id BIGINT PRIMARY KEY AUTO_INCREMENT, name varchar2(2000))
